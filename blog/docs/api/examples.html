Code-Beispiele

Diese Seite ist auf Tiefe 3 (/docs/api/examples.html).

JavaScript / Node.js

const scanner = require('@a11yplan/scanner');

async function runScan() {
  const results = await scanner.scan({
    url: 'https://example.com',
    level: 'AA'
  });
  console.log(results);
}

Python

import requests

response = requests.post(
    'https://api.a11yplan.com/v1/scan',
    json={'url': 'https://example.com'}
)
print(response.json())
